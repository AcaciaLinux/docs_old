#Install Grub in EFI Mode to EFI-Boot-Partiton:
- mount --bind dev, sys... to chroot

mount -v --bind /dev $LFS/dev
mount --bind /dev/pts $LFS/dev/pts
mount -t proc proc $LFS/proc
mount -t sysfs sysfs $LFS/sys
mount -t tmpfs tmpfs $LFS/run


- mount fat32 formatted EFI Partition to /boot/efi

- mount sysvars
mountpoint /sys/firmware/efi/efivars || mount -v -t efivarfs efivarfs /sys/firmware/efi/efivars

- install grub
grub-install --botloader-id=<ID> --recheck

- create grub.cfg

# Begin /boot/grub/grub.cfg
set default=0
set timeout=5

insmod part_gpt

# !! DONT FORGET TO CHANGE TO CORRECT FS !!
insmod ext2

# !! DONT FORGET TO SET CORRECT ROOTDISK IDS !!
set root=(hd0,2)

if loadfont /boot/grub/fonts/unicode.pf2; then
  set gfxmode=auto
  insmod all_video
  terminal_output gfxterm
fi

menuentry "GNU/Linux, Linux 5.10.17-lfs-10.1"  {
  linux   /boot/vmlinuz-5.10.17-lfs-10.1 root=/dev/sda2 ro
}

menuentry "Firmware Setup" {
  fwsetup
}
